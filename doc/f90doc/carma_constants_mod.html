Description of Variables

T0

 
real (kind=f), public, parameter :: T0 = 273.16_f
Define triple-point temperature (K)

PI

 
real (kind=f), public, parameter :: PI = 3.14159265358979_f

DEG2RAD

 
real (kind=f), public, parameter :: DEG2RAD = PI / 180._f

RAD2DEG

 
real (kind=f), public, parameter :: RAD2DEG = 180._f / PI

GCONST

 
real (kind=f), public, parameter :: GCONST = 6.67408e-8_f
Define gravitational constant [ cm^3 / g / s ]

AVG

 
real (kind=f), public, parameter :: AVG = 6.02252e+23_f
Define avogadro's number [ # particles / mole ]

BK

 
real (kind=f), public, parameter :: BK = 1.38054e-16_f
Define Boltzmann's constant [ erg / deg_K ]

MPROT

 
real (kind=f), public, parameter :: MPROT = 1.6726219e-24_f
Define proton mass [ g ]

ALOS

 
real (kind=f), public, parameter :: ALOS = 2.68719e+19_f
Define Loschmidt's number [ mole / cm^3, @ STP ]

PREF

 
real (kind=f), public, parameter :: PREF = 1000.e+3_f
Define reference pressure, e.g. for potential temp calcs [ dyne / cm^2 ]

RMB2CGS

 
real (kind=f), public, parameter :: RMB2CGS = 1000.e+0_f
Define conversion factor for mb to cgs [ dyne / cm^2 ] units

RPA2CGS

 
real (kind=f), public, parameter :: RPA2CGS = 10.e+0_f
Define conversion factor for Pa to cgs [ dyne / cm^2 ] units

RM2CGS

 
real (kind=f), public, parameter :: RM2CGS = 100.0_f
Define conversion factor for m to cgs [ cm ] units

RGAS

 
real (kind=f), public, parameter :: RGAS = 8.31430e+07_f
Define universal gas constant [ erg / deg_K / mole ]

EV2ERG

 
real (kind=f), public, parameter :: EV2ERG = 1.6e-12_f
Define electronvolt in [ erg ]

COLINT

 
real (kind=f), public, parameter :: COLINT = 1._f
Value of Collision Integral

IT

 
integer, public, parameter :: IT = 1
The dimension of THETD, ELTRMX, CSTHT, PI, TAU, SI2THT. IT must correspond exactly to the second dimension of ELTRMX.

CARMA_NAME_LEN

 
integer, public, parameter :: CARMA_NAME_LEN = 255
String length of names

CARMA_SHORT_NAME_LEN

 
integer, public, parameter :: CARMA_SHORT_NAME_LEN = 6
String length of short names

CAM_FILL

 
integer, public, parameter :: CAM_FILL = -999
Fill value indicating no value is being returned

SMALL_PC

 
real (kind=f), public, parameter :: SMALL_PC = 1e-150_f
Define small particle number concentration [ # / x_units / y_units / z_units ]

FEW_PC

 
real (kind=f), public, parameter :: FEW_PC = SMALL_PC * 1e6_f
Define particle number concentration [ # / ? ] used to decide whether to bypass microphysical processes.

Set it to SMALL_PC/xmet/ymet/zmet to never bypass the calculations.

FIX_COREF

real (kind=f), public, parameter :: FIX_COREF = 1e-50_f
Define core fraction (for core mass and second moment) used when particle number concentrations are limited to SMALL_PC

CLDFRC_MIN

 
real (kind=f), public, parameter :: CLDFRC_MIN = 1e-4_f
Minimum Cloud Fraction

CLDFRC_INCLOUD

 
real (kind=f), public, parameter :: CLDFRC_INCLOUD = 0.10_f
Incloud Cloud Fraction Threshold for statistics
